you like some challenge?


            
document.getElementById("submit_email").addEventListener("click", function(){
    var email = document.getElementById('email').value;
    LogonClient.getProviders(email, function(res) {
        var p1 = Object.assign({}, res);
        var p2 = Object.assign({
            theme: 'icon'
        }, res);
        LogonClient.ui.button('logonlabs6', p1);
        LogonClient.ui.button('logonlabs7', p2);
        document.getElementById('email_valid').innerText =
            "Here are list of providers this email can login: " + email;
    });
});
